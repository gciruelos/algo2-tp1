\begin{tad}{\tadNombre{Conexion}}
 
 \tadGeneros{conexion}
 \tadExporta {conexion, observadores, generadores, otrasoperaciones}
 \tadUsa {compu, bool}
 
 \tadObservadores
 \tadOperacion{computadorasQueConecta}{conexion}{conj(compu)}{}
 
 \tadGeneradores
 %crearOperacion necesitaria ver tambien que no exista ya una conexion de la compu 1 usando la interfaz 1, y lo mismo para c2 con i2... quiza habria que crear algun otro observador.
 \tadOperacion{crearConexion}{tupla(compu/c1, interfaz/i1), tupla(compu/c2, interfaz/i2)}{conexion}{tieneInterfaz(c1,i1) $\wedge$ tieneInterfaz(c2,i2)}
 
 \tadOtrasOperaciones
 
 \tadOperacion{conectaA}{conexion, compu}{bool}{}
 \tadOperacion{laOtraCompu}{conexion c, compu c1}{compu}{c1 \in compusQueConecta(c)}
 \tadOperacion{estanConectadas}{conexion, compu, compu}{bool}{}
 
 \tadAxiomas [\paratodo{compu}{c1,c2}, \paratodo{conexion}{conexion}, \paratodo{interfaz}{i1,i2}]
 
 \tadAxioma{computadorasQueConecta(crearConexion (c1,i1) (c2,i2))}{Ag(c1, Ag(c2, Vacio))}
 
 \tadAxioma{conectaA(conexion, c1)}{c1 \in (computadorasQueConecta(conexion))}
 
 \tadAxioma{laOtraCompu(conexion, c1, c2)}{dameUno((computadorasQueConecta(conexion)) - c1)}
 
 \tadAxioma{estanConectadas(conexion, c1, c2)}{conectaA(conexion,c1) && conectaA(conexion,c2)}

\end{tad}

\begin{tad}{\tadNombre{Paquete}}
\tadIgualdadObservacional{p}{p'}{paq}{destinoPaq($p$) \igobs destinoPaq($p'$) $\wedge$\\
									 origenPaq($p$) \igobs origenPaq($p'$) $\wedge$\\
									 priorPaq($p$) \igobs priorPaq($p'$) $\wedge$\\
									 datoPaq($p$) \igobs datoPaq($p'$) $\wedge$\\
									 idPaq($p$) \igobs idPaq($p'$)}

\tadGeneros{paq}
\tadExporta {paq, generadores, observadores, $<$}
\tadUsa {Nat, Bool, Computadora, String}


\tadObservadores
  \tadAlinearFunciones{destinoPaq}{paq}
  \tadOperacion{origenPaq}{paq}{compu}{}
  \tadOperacion{destinoPaq}{paq}{compu}{}
  \tadOperacion{priorPaq}{paq}{nat}{}
  \tadOperacion{datoPaq}{paq}{string}{}
  \tadOperacion{idPaq}{paq}{nat}{}

\tadGeneradores

  \tadAlinearFunciones{nuevoPaquete}{compu, compu, nat , string, nat}
  \tadOperacion{nuevoPaquete}{compu, compu, nat , string, nat}{paq}{}

\tadOtrasOperaciones
\tadOperacion{$\puntito < \puntito$}{paq, paq}{bool}{}


\tadAxiomas[\paratodo{paq}{p,p'}, \paratodo{nat}{prior, id}, \paratodo{compu}{origen, destino}, \paratodo{string}{s}]
  \tadAlinearAxiomas{destinoPaq(nuevoPaquete(origen, destino, prior, s, id))}
  \tadAxioma{origenPaq(nuevoPaquete(origen, destino, prior, s, id))}{origen}
  \tadAxioma{destinoPaq(nuevoPaquete(origen, destino, prior, s, id))}{destino}
  \tadAxioma{priorPaq(nuevoPaquete(origen, destino, prior, s, id))}{prior}
  \tadAxioma{datoPaq(nuevoPaquete(origen, destino, prior, s, id))}{dato}
  \tadAxioma{idPaq(nuevoPaquete(origen, destino, prior, s, id))}{id}
  \tadAxioma{p $<$ p'}{priorPaq(p) $<$ priorPaq(p')}

\end{tad}
